<div id="raw-poem" style="display:none;">
للعالم الأعلى تتوق الروحُ = وبكلّ سرٍ في الفؤاد تبوحُ ,
تشقى بأغلال الحياة يحثها = أملٌ وإن زاد الأسى ستنوحُ ,
دارتْ على الرغبات دورة غافلٍ = هيمانَ في كل الفجاج يروحُ ,
الجوهر الطينيُّ قص جناحها = عن كل نجمٍ في السماء يلوحُ ,
ومن المصاعب ما يلذُّ إلى الفتى = ومن النعيم مصائبٌ وجروحُ
</div>

<div id="rendered-poem" class="poem-wrapper"></div>

<script>
    const rawData = document.getElementById('raw-poem').innerText;
    const renderedContainer = document.getElementById('rendered-poem');
    const verses = rawData.split(',');

    let htmlOutput = '';
    verses.forEach(verse => {
        if (!verse.trim()) return;
        const shatrs = verse.split('=');
        if (shatrs.length === 2) {
            htmlOutput += `<div class="bayt">
                <div class="shatr">${shatrs[0].trim()}</div>
                <div class="shatr">${shatrs[1].trim()}</div>
            </div>`;
        } else {
             htmlOutput += `<div class="bayt"><div class="shatr" style="text-align:center; width:100%">${verse.trim()}</div></div>`;
        }
    });
    renderedContainer.innerHTML = htmlOutput;
</script>
